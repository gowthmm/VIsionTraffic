    <script>
        // Global variables
        const uploadedFiles = [];
        const selectedVideos = [];
        const activeProcessors = {};
        let processingActive = false;
        let pollingStopped = {}; // global list, keeps track of polling for each video
        let vehicleCounts = {}; // Global store of vehicle count.

        // Create video processor template
        function createVideoProcessor(id, videoPath) {
    const filename = videoPath.split('/').pop();
    const processorDiv = document.createElement('div');
    processorDiv.className = 'video-processor';
    processorDiv.id = `processor-${id}`;
    processorDiv.innerHTML = `
        <div class="video-processor-title" title="${filename}">${filename}</div>
        <div class="progress-container">
            <div class="progress-bar" id="progressBar-${id}">0%</div>
        </div>
        <div class="frame-container">
            <div class="overlay" id="overlay-${id}"></div>
            <img id="currentVideo-${id}" class="frame" src="" alt=""/>
            <div class="signal-info">
                <div class="traffic-signal" id="trafficSignal-${id}"></div>
                <div class="timer" id="timer-${id}">--:--</div>
            </div>
        </div>
        <div class="detection-info">
            <div class="detection-box">
                <h4>Objects</h4>
                <div id="objectList-${id}"></div>
            </div>
            <div class="detection-box">
                <h4>Vehicles</h4>
                <div id="vehicleCount-${id}">0</div>
            </div>
        </div>
        <div class="processor-status" id="status-${id}">Ready</div>
    `;
    return processorDiv;
        }


        // Drag and drop functionality
        const dropArea = document.getElementById('dropArea');

        ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
            dropArea.addEventListener(eventName, preventDefaults, false);
        });

        function preventDefaults(e) {
            e.preventDefault();
            e.stopPropagation();
        }

        ['dragenter', 'dragover'].forEach(eventName => {
            dropArea.addEventListener(eventName, highlight, false);
        });

        ['dragleave', 'drop'].forEach(eventName => {
            dropArea.addEventListener(eventName, unhighlight, false);
        });

        function highlight() {
            dropArea.classList.add('highlight');
        }

        function unhighlight() {
            dropArea.classList.remove('highlight');
        }

        dropArea.addEventListener('drop', handleDrop, false);

        function handleDrop(e) {
            const dt = e.dataTransfer;
            const files = dt.files;
            handleFiles(files);
        }

        // File input handler
        document.getElementById('fileInput').addEventListener('change', function (e) {
            handleFiles(this.files);
        });

        function handleFiles(files) {
            if (files.length === 0) return;

            const formData = new FormData();
            for (let i = 0; i < files.length; i++) {
                formData.append('videos', files[i]);
            }

            updateStatus("Uploading videos...");
            logMessage("Starting upload...");

            fetch('/upload', {
                    method: 'POST',
                    body: formData
                })
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`HTTP error! Status: ${response.status}`);
                    }
                    return response.json();
                })
                .then(data => {
                    updateStatus(`Uploaded ${data.files.length} videos successfully`);
                    logMessage(`Upload complete: ${data.files.map(f => f.split('/').pop()).join(', ')}`);

                    // Add uploaded files to our list
                    data.files.forEach(file => {
                        if (!uploadedFiles.includes(file)) {
                            uploadedFiles.push(file);
                        }
                    });

                    updateFileList();
                })
                .catch(error => {
                    console.error('Error:', error);
                    updateStatus(`Error uploading videos: ${error.message}`);
                    logMessage(`Upload error: ${error.message}`);
                });
        }

        function updateFileList() {
            const fileList = document.getElementById('fileList');
            fileList.innerHTML = '';

            uploadedFiles.forEach(file => {
                const filename = file.split('/').pop();
                const fileItem = document.createElement('div');
                fileItem.className = 'file-item';

                const isSelected = selectedVideos.includes(file);

                fileItem.innerHTML = `
                    <span>${filename}</span>
                    <span class="file-select" data-path="${file}">${isSelected ? 'Deselect' : 'Select'}</span>
                `;
                fileList.appendChild(fileItem);
            });

            // Add event listeners to select/deselect
            document.querySelectorAll('.file-select').forEach(elem => {
                elem.addEventListener('click', function () {
                    const path = this.getAttribute('data-path');
                    toggleVideoSelection(path);
                });
            });

            updateProcessButton();
        }

        function toggleVideoSelection(path) {
            const index = selectedVideos.indexOf(path);
            if (index === -1) {
                // Not selected, add it if we have less than 4
                if (selectedVideos.length < 4) {
                    selectedVideos.push(path);
                    logMessage(`Selected video: ${path.split('/').pop()}`);
                } else {
                    updateStatus("Maximum of 4 videos can be selected at once");
                }
            } else {
                // Already selected, remove it
                selectedVideos.splice(index, 1);
                logMessage(`Deselected video: ${path.split('/').pop()}`);
            }

            updateFileList();
            updateProcessButton();
        }

        function updateProcessButton() {
            const processBtn = document.getElementById('processBtn');
            processBtn.disabled = selectedVideos.length === 0 || processingActive;

            if (selectedVideos.length > 0) {
                updateStatus(`${selectedVideos.length} video(s) selected. Ready to process.`);
            } else {
                updateStatus("Select up to 4 videos to process simultaneously.");
            }
        }

        // Process videos button
        document.getElementById('processBtn').addEventListener('click', function () {
            if (selectedVideos.length === 0) {
                updateStatus("Please select at least one video to process");
                return;
            }

            // Set processing state
            processingActive = true;
            document.getElementById('processBtn').disabled = true;
            document.getElementById('stopBtn').disabled = false;

            // Clear previous processing area
            document.getElementById('videoGrid').innerHTML = '';

            // Reset polling stopped for the processors.
            pollingStopped = {};

            // Setup processors for each selected video
            let promises = []; // track promises, so that we can choose to play a video

            selectedVideos.forEach((videoPath, index) => {
                const processorId = `video-${index}-${Date.now()}`; // Unique ID
                const processorElement = createVideoProcessor(processorId, videoPath);
                document.getElementById('videoGrid').appendChild(processorElement);

                //Perform initial analysis after a delay
                const promise = new Promise((resolve) => {
                    setTimeout(() => {
                        performInitialAnalysis(processorId, videoPath)
                            .then(() => {
                                resolve(processorId); // Resolve with the processorId
                            });
                    }, 3000); // 3-second delay
                });
                promises.push(promise)

                // Start processing for this video
                const interval = parseInt(document.getElementById('frameInterval').value);
                logMessage(`Starting processor ${index} for ${videoPath.split('/').pop()}`);
                startVideoProcessing(processorId, videoPath, interval); //New start function
            });

            Promise.all(promises).then(processorIds => {
                //All video's initial detections are done.
                vehicleCounts = {};
                processorIds.forEach(pid => {
                    let el = document.getElementById(`vehicleCount-${pid}`)
                    vehicleCounts[pid] = parseInt(el.textContent);
                })

                //Find the one with the biggest id!
                max_vehicle_id = Object.keys(vehicleCounts).reduce(function (a, b) {
                    return vehicleCounts[a] > vehicleCounts[b] ? a : b
                });
                //After all of the initial analysis is done, play only the video.
                processorIds.forEach(pid => {
                    //Pause video except for the largest one
                    get_paused_state(pid)

                })
            })

            updateStatus(`Processing ${selectedVideos.length} videos simultaneously`);
        });

        // Stop processing button
        document.getElementById('stopBtn').addEventListener('click', function () {
            stopAllProcessing();
            updateStatus("Processing stopped by user");
            logMessage("All processing stopped by user");
        });

        function stopAllProcessing() {
            processingActive = false;

            // Stop all individual processors
            Object.keys(activeProcessors).forEach(index => {
                activeProcessors[index].processing = false;
                stopPolling(index); //Also stop polling!

                //Pause the videos

            });

            document.getElementById('processBtn').disabled = false;
            document.getElementById('stopBtn').disabled = true;
        }

        function startVideoProcessing(processorId, videoPath, interval) {
            fetch('/process_video', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        video_path: videoPath,
                        interval: interval,
                        processor_id: processorId
                    })
                })
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`HTTP error! Status: ${response.status}`);
                    }
                    return response.json();
                })
                .then(data => {
                    if (data.error) {
                        throw new Error(data.error);
                    }
                    const totalFrames = data.total_frames;
                    const fps = data.fps;
                    logMessage(`Video ${processorId} has ${totalFrames} frames at ${fps} FPS`);
                    //Start polling for frame data
                    pollForFrameData(processorId, totalFrames);
                })
                .catch(error => {
                    console.error(`Error starting processor ${processorId}:`, error);
                    updateProcessorStatus(processorId, `Error: ${error.message}`);
                    logMessage(`Error in processor ${processorId}: ${error.message}`);

                });
        }

        function stopPolling(processorId) {
            pollingStopped[processorId] = true;
        }

        function startPolling(processorId, totalFrames) {
            pollingStopped[processorId] = false;
        }

        //Poller function that gets current frame data, and its status.
        function pollForFrameData(processorId, totalFrames) {
            pollingStopped[processorId] = false; //Allow running by default for this loop

            function fetchUpdates() {
                if (pollingStopped[processorId]) {
                    console.log(`Polling stopped for ${processorId}`);
                    return; // Stop polling if flagged
                }
                
                // Get status first
                fetch(`/get_processing_status/${processorId}`)
                    .then(response => {
                        if (!response.ok) {
                            throw new Error(`HTTP error! Status: ${response.status}`);
                        }
                        return response.json();
                    })
                    .then(statusData => {
                        if (statusData.processing && !statusData.complete) {
                            // Fetch image data only if processing is active
                            fetch(`/get_frame_data/${processorId}`)
                                .then(response => {
                                    if (!response.ok) {
                                        throw new Error(`HTTP error! Status: ${response.status}`);
                                    }
                                    return response.json();
                                })
                                .then(frameData => {
                                    if (frameData && frameData.image) {
                                        // Update the image source with new frame
                                        const videoElement = document.getElementById(`currentVideo-${processorId}`);
                                        videoElement.src = `data:image/jpeg;base64,${frameData.image}`;
                                        
                                        // Update object list
                                        const objectList = document.getElementById(`objectList-${processorId}`);
                                        objectList.innerHTML = "";
                                        for (const [obj, count] of Object.entries(frameData.objects)) {
                                            objectList.innerHTML += `<div>${obj}: ${count}</div>`;
                                        }
                                        
                                        // Update vehicle count
                                        const vehicleCount = frameData.vehicles || 0;
                                        vehicleCounts[processorId] = vehicleCount;
                                        document.getElementById(`vehicleCount-${processorId}`).textContent = vehicleCount;
                                        
                                        // Update progress bar
                                        const progress = Math.min(100, Math.round((statusData.progress / totalFrames) * 100));
                                        updateProcessorProgress(processorId, progress);
                                        
                                        // Update traffic signal display
                                        updateTrafficSignalDisplay(processorId, frameData.signal);
                                         const timerElement = document.getElementById(`timer-${processorId}`);
                                if (frameData.signal === "green" && frameData.remaining_time !== undefined) {
                                    const seconds = Math.round(frameData.remaining_time);
                                    timerElement.textContent = `${seconds}s remaining`;
                                    timerElement.style.color = "green";
                                } else if (frameData.allocated_time !== undefined) {
                                    timerElement.textContent = `Allocated: ${Math.round(frameData.allocated_time)}s`;
                                    timerElement.style.color = "gray";
                                } else {
                                    timerElement.textContent = "--:--";
                                }
                                        // If signal is green, ensure video is not paused
                                        if (frameData.signal === "green" && statusData.paused) {
                                            set_paused_state(processorId, false);
                                            logMessage(`Auto-resumed ${processorId} (green signal)`);
                                        }
                                    }
                                    
                                    // Continue polling regardless of result
                                    setTimeout(fetchUpdates, 500);
                                })
                                .catch(error => {
                                    console.error(`Error getting frame data for ${processorId}:`, error);
                                    setTimeout(fetchUpdates, 1000); // Try again after error with longer delay
                                });
                        } else {
                            // Processing complete or stopped
                            updateProcessorStatus(processorId, statusData.complete ? "Complete!" : "Stopped.");
                            updateTrafficSignalDisplay(processorId, "red");
                        }
                    })
                    .catch(error => {
                        console.error(`Error getting processing status for ${processorId}:`, error);
                        setTimeout(fetchUpdates, 1000); // Try again after error with longer delay
                    });
            }

            // Start the polling immediately
            fetchUpdates();
        }

        //A dedicated function for intial analysis after timeout
        function performInitialAnalysis(processorId, videoPath) {
            return fetch('/initial_analysis', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    },
                    body: JSON.stringify({
                        video_path: videoPath,
                        processor_id: processorId
                    })
                })
                .then(response => {
                    if (!response.ok) {
                        throw new Error(`HTTP error! Status: ${response.status}`);
                    }
                    return response.json();
                })
                .then(data => {
                    if (data.error) {
                        throw new Error(data.error);
                    }
                    const vehicleCount = data.vehicle_count;
                    document.getElementById(`vehicleCount-${processorId}`).textContent = vehicleCount; //Set the count
                    logMessage(`Initial analysis complete for ${processorId}: ${vehicleCount} vehicles`);
                })
                .catch(error => {
                    console.error(`Error during initial analysis for ${processorId}:`, error);
                    updateProcessorStatus(processorId, `Initial analysis error: ${error.message}`);
                });
        }

        function updateTrafficSignalDisplay(processorId, signalColor) {
            const trafficSignalElement = document.getElementById(`trafficSignal-${processorId}`);
            trafficSignalElement.className = 'traffic-signal'; // Reset classes
            trafficSignalElement.classList.add(signalColor);

            const overlayElement = document.getElementById(`overlay-${processorId}`);
            overlayElement.className = 'overlay';
            overlayElement.classList.add(signalColor) //Signal to overlay class, the color.

             // Handle the signal change
            handleSignalChange(processorId, signalColor);
        }


        function updateProcessorProgress(processorId, percent) {
            const progressBar = document.getElementById(`progressBar-${processorId}`);
            if (progressBar) {
                progressBar.style.width = percent + '%';
                progressBar.textContent = percent + '%';
            }
        }

        function updateProcessorStatus(processorId, message) {
            const statusElem = document.getElementById(`status-${processorId}`);
            if (statusElem) {
                statusElem.textContent = message;
            }
        }

        function updateStatus(message) {
            document.getElementById('status').textContent = message;
        }

        function logMessage(message) {
            const log = document.getElementById('log');
            const timestamp = new Date().toLocaleTimeString();
            log.innerHTML += `<div>[${timestamp}] ${message}</div>`;
            log.scrollTop = log.scrollHeight;
        }

       function set_paused_state(processorId, paused){
            logMessage(`Setting ${processorId} paused state to ${paused}`);
            
            fetch(`/set_paused/${processorId}/${paused}`, {
                method: 'GET'
            })
            .then(response => {
                if (!response.ok) {
                    throw new Error(`HTTP error! Status: ${response.status}`);
                }
                return response.json();
            })
            .then(data => {
                if (data.error) {
                    throw new Error(data.error);
                }
                logMessage(`Successfully set ${processorId} paused state to ${paused}`);
            })
            .catch(error => {
                console.error(`Error setting paused state for ${processorId}:`, error);
                logMessage(`Error setting paused state: ${error.message}`);
            });
        }

        function handleSignalChange(processorId, signal) {
            if (signal === "green") {
                // Resume video processing
                set_paused_state(processorId, false);
                // Start polling for frames
                startPolling(processorId);
                
                // Update visual indicators
                const videoElement = document.getElementById(`currentVideo-${processorId}`);
                videoElement.classList.remove('paused');
                
                logMessage(`${processorId} received GREEN signal - traffic flowing`);
            } else {
                // Pause video processing
                set_paused_state(processorId, true);
                
                // Update visual indicators
                const videoElement = document.getElementById(`currentVideo-${processorId}`);
                videoElement.classList.add('paused');
                
                logMessage(`${processorId} received RED signal - traffic stopped`);
            }
        }
